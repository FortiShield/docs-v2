Override the default [Storage](/storage) interface used by the SDK.

Allows you to create an instance of `EgiftcardStorage` with your own customized config, and pass it to the SDK.

<Callout variant='info'>

This requires the `@egiftcard/storage` package to be installed.

[Learn more about Storage](/storage)

</Callout>

```jsx
import { EgiftcardProvider } from "@egiftcard/react-native";
import {
	EgiftcardStorage,
	StorageDownloader,
	IpfsUploader,
} from "@egiftcard/storage";

// Configure a custom EgiftcardStorage instance
const storage = new EgiftcardStorage({
	uploader: new IpfsUploader(),
	downloader: new StorageDownloader(),
	gatewayUrls: {
		"ipfs://": [
			"https://gateway.ipfscdn.io/ipfs/",
			"https://cloudflare-ipfs.com/ipfs/",
			"https://ipfs.io/ipfs/",
		],
	},
});

// Provide the custom storage instance to the SDK
function MyApp() {
	return (
		<EgiftcardProvider storageInterface={storage}>
			<YourApp />
		</EgiftcardProvider>
	);
}
```
